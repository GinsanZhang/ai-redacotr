## 3. 核心算法详解

### 3.1 云端视觉识别算法

```python
def detect_by_cloud_vision(image: np.ndarray, progress_cb=None) -> list:
    """
    使用云端多模态模型进行 OCR
    
    算法步骤:
    1. 图片压缩（最大边限制 1800px）
    2. Base64 编码
    3. 构建 Prompt 要求返回结构化数据
    4. 流式请求 API
    5. 解析 JSON 结果
    6. 坐标反归一化（[0,1000] -> 原图像素）
    
    参数:
        image: OpenCV BGR 格式图片
        progress_cb: 进度回调函数
    
    返回:
        List[dict]: OCR 结果列表
    
    异常:
        RuntimeError: API 调用失败
    """
    # 1. 压缩图片
    h, w = image.shape[:2]
    scale = min(1.0, 1800.0 / max(h, w))
    resized = cv2.resize(image, (int(w * scale), int(h * scale))) if scale < 1.0 else image
    
    # 2. 编码为 JPEG
    ok, buf = cv2.imencode(".jpg", resized, [int(cv2.IMWRITE_JPEG_QUALITY), 85])
    if not ok: return []
    
    # 3. Base64 编码
    b64 = base64.b64encode(buf.tobytes()).decode("utf-8")
    
    # 4. 构建 Prompt
    prompt = """你是一个高精度的 OCR 助手。请识别图片中所有的文字块。
    重要：bbox 坐标必须是基于图片尺寸归一化到 [0, 1000] 范围内的整数。
    格式：{"blocks":[{"text":"...","bbox":[xmin,ymin,xmax,ymax]}]}"""
    
    # 5. 调用 API (流式)
    body = {
        "model": CONFIG["cloud_vision_model"],
        "messages": [{"role": "user", "content": [
            {"type": "text", "text": prompt},
            {"type": "image_url", "image_url": {"url": f"data:image/jpeg;base64,{b64}"}}
        ]}],
        "temperature": 0.0
    }
    
    # 6. 解析结果
    content = request_chat_stream_content(...)
    data = json.loads(content)
    
    # 7. 坐标反归一化
    for block in data["blocks"]:
        nx1, ny1, nx2, ny2 = block["bbox"]
        x1 = int(nx1 * w / 1000.0)
        y1 = int(ny1 * h / 1000.0)
        x2 = int(nx2 * w / 1000.0)
        y2 = int(ny2 * h / 1000.0)
        result.append({
            "text": block["text"],
            "bbox": [max(0, x1), max(0, y1), min(w, x2), min(h, y2)],
            "conf": block.get("conf", 1.0)
        })
    
    return result
```

### 3.2 规则匹配算法

```python
def detect_by_rules(ocr_blocks: list) -> list:
    """
    使用正则表达式匹配敏感信息
    
    算法步骤:
    1. 遍历每个 OCR 结果
    2. 对每个模式进行正则匹配
    3. 首次匹配即记录，跳出内层循环
    
    时间复杂度: O(n*m) 
        n: OCR 块数量
        m: 模式数量
    
    参数:
        ocr_blocks: OCR 结果列表
    
    返回:
        List[dict]: 命中结果列表
    """
    hits = []
    for block in ocr_blocks:
        text = block["text"]
        for label, pattern in PATTERNS.items():
            if re.search(pattern, text):
                hits.append({
                    "bbox": block["bbox"],
                    "label": label,
                    "text": text,
                    "source": "rule"
                })
                break  # 一个块只匹配一次
    return hits
```

```python
def detect_by_ai(ocr_blocks: list, progress_cb=None) -> list:
    """
    使用大模型识别非标准隐私信息
    
    算法步骤:
    1. 过滤已由规则识别的文本
    2. 构建 Prompt 要求 LLM 分析
    3. 流式调用 API
    4. 解析 JSON 结果
    5. 映射回原始 OCR 块
    
    优化点:
    - 仅分析未被规则识别的内容
    - 使用结构化输出降低解析难度
    
    参数:
        ocr_blocks: 剩余 OCR 结果
        progress_cb: 进度回调
    
    返回:
        List[dict]: AI 识别结果
    """
    if not CONFIG["ai_enabled"] or not ocr_blocks:
        return []
    
    # 构建结构化输入
    items = [{"id": i, "text": b["text"]} for i, b in enumerate(ocr_blocks)]
    
    prompt = f"""你是一个隐私保护专家。请从以下 OCR 文字块中识别个人隐私或敏感信息。
    
    待分析数据：{json.dumps(items, ensure_ascii=False)}
    
    识别范围：姓名、详细地址、职位、社交账号、企业名称等。
    过滤规则：手机号、身份证等已由规则处理，请忽略。不要标记公共按钮。
    
    输出要求：返回 JSON: {{"sensitive": [{{"id": 0, "reason": "姓名"}}]}}。"""
    
    # 调用 API
    content, status = request_chat_stream_content(...)
    
    if status == 200:
        # 提取 JSON
        s, e = content.find('{'), content.rfind('}') + 1
        data = json.loads(content[s:e])
        
        # 映射结果
        hits = []
        for item in data.get("sensitive", []):
            idx = item.get("id")
            if 0 <= idx < len(ocr_blocks):
                hits.append({
                    "bbox": ocr_blocks[idx]["bbox"],
                    "label": item.get("reason", "AI识别"),
                    "text": ocr_blocks[idx]["text"],
                    "source": "ai"
                })
        return hits
    
    return []
```

### 3.4 马赛克应用算法

```python
def apply_mosaic(image: np.ndarray, bbox: list, style: str, strength: int) -> np.ndarray:
    """
    对指定区域应用马赛克
    
    算法步骤:
    1. 扩展区域边界（padding=4px）
    2. 根据样式选择处理方式
    3. 修改原图对应区域
    
    样式说明:
    - blur: 高斯模糊，核大小 = max(strength|1, 21)
    - pixel: 像素化，块大小 = max(strength//3, 8)
    - block: 纯黑填充
    
    参数:
        image: 原图 (BGR)
        bbox: [x1, y1, x2, y2]
        style: blur|pixel|block
        strength: 强度 5-50
    
    返回:
        修改后的图片
    """
    x1, y1, x2, y2 = bbox
    p = 4  # padding
    
    # 边界扩展和裁剪
    x1 = max(0, x1 - p)
    y1 = max(0, y1 - p)
    x2 = min(image.shape[1], x2 + p)
    y2 = min(image.shape[0], y2 + p)
    
    if x2 <= x1 or y2 <= y1:
        return image
    
    region = image[y1:y2, x1:x2]
    
    if style == "blur":
        # 高斯模糊
        k = max(strength | 1, 21)  # 确保奇数且 >=21
        image[y1:y2, x1:x2] = cv2.GaussianBlur(region, (k, k), 0)
    
    elif style == "pixel":
        # 像素化
        h, w = region.shape[:2]
        block = max(strength // 3, 8)
        small = cv2.resize(region, (max(1, w // block), max(1, h // block)), 
                          interpolation=cv2.INTER_LINEAR)
        image[y1:y2, x1:x2] = cv2.resize(small, (w, h), 
                                        interpolation=cv2.INTER_NEAREST)
    
    elif style == "block":
        # 纯黑遮罩
        image[y1:y2, x1:x2] = 0
    
    return image
```

## 4. UI 渲染流程

### 4.1 画布渲染流程

```python
def _render(self):
    """
    渲染画布
    
    渲染步骤:
    1. 应用马赛克到原图
    2. 缩放图片到显示尺寸
    3. 计算居中偏移量
    4. 创建全尺寸 Pixmap
    5. 绘制图片
    6. 绘制打码区域框
    7. 绘制标签
    8. 绘制拖拽预览框
    """
    if self.original_cv is None:
        return
    
    # 1. 应用马赛克
    img = self.original_cv.copy()
    for r in self.regions:
        img = apply_mosaic(img, r["bbox"], CONFIG["mosaic_style"], 
                          CONFIG["mosaic_strength"])
    
    # 2. 计算显示尺寸
    h, w = img.shape[:2]
    nw = int(w * self.display_scale)
    nh = int(h * self.display_scale)
    
    # 3. 计算偏移（居中显示）
    offset_x = max(0, (self.width() - nw) // 2)
    offset_y = max(0, (self.height() - nh) // 2)
    self.image_offset = (offset_x, offset_y)
    
    # 4. 创建全尺寸 Pixmap
    pix = QPixmap(self.width(), self.height())
    pix.fill(GLOBAL_TRANSPARENT)
    
    ptr = QPainter(pix)
    ptr.setRenderHint(PAINTER_ANTIALIAS)
    
    # 5. 绘制图片
    disp = cv2.resize(img, (nw, nh))
    rgb = cv2.cvtColor(disp, cv2.COLOR_BGR2RGB)
    img_pix = QPixmap.fromImage(QImage(rgb.data, nw, nh, 3 * nw, IMAGE_FORMAT_RGB888))
    ptr.drawPixmap(offset_x, offset_y, img_pix)
    
    # 6. 绘制区域框
    for i, r in enumerate(self.regions):
        # 坐标转换：原图 -> 显示 -> 偏移
        bbox = r["bbox"]
        x1 = int(bbox[0] * self.display_scale) + offset_x
        y1 = int(bbox[1] * self.display_scale) + offset_y
        x2 = int(bbox[2] * self.display_scale) + offset_x
        y2 = int(bbox[3] * self.display_scale) + offset_y
        
        # 颜色根据来源
        if r.get("source") == "ai":
            col = QColor(255, 160, 0, 180)
        else:
            col = QColor(255, 80, 80, 180)
        
        # 悬停高亮
        if i == self.hover_region:
            col = QColor(255, 50, 50, 220)
        
        ptr.setPen(QPen(col, 2))
        ptr.setBrush(QBrush(QColor(col.red(), col.green(), col.blue(), 30)))
        ptr.drawRect(x1, y1, x2 - x1, y2 - y1)
        
        # 绘制标签
        if r.get("label"):
            ptr.setPen(QPen(GLOBAL_WHITE))
            ptr.setBrush(QBrush(col))
            fm = ptr.fontMetrics()
            tw = fm.horizontalAdvance(r["label"]) + 8
            th = fm.height() + 4
            ptr.drawRect(x1, max(offset_y, y1 - th), tw, th)
            ptr.drawText(x1 + 4, max(th + offset_y, y1) - 3, r["label"])
    
    # 7. 绘制拖拽预览
    if self.drag_start and self.drag_current:
        x1 = min(self.drag_start.x(), self.drag_current.x())
        y1 = min(self.drag_start.y(), self.drag_current.y())
        x2 = max(self.drag_start.x(), self.drag_current.x())
        y2 = max(self.drag_start.y(), self.drag_current.y())
        ptr.setPen(QPen(QColor(100, 200, 255), 2, PEN_DASH_LINE))
        ptr.setBrush(QBrush(QColor(100, 200, 255, 40)))
        ptr.drawRect(x1, y1, x2 - x1, y2 - y1)
    
    ptr.end()
    self.setPixmap(pix)
```

```
坐标系统:
1. 屏幕坐标系: 鼠标事件返回的坐标，相对于 QLabel 左上角
2. 显示坐标系: 屏幕坐标 - image_offset，相对于图片显示区域
3. 原图坐标系: 显示坐标 / display_scale，相对于原始图片
```

### 5.2 坐标转换公式

```python
# 鼠标事件坐标 -> 原图坐标（存储）
def screen_to_original(screen_x, screen_y, offset_x, offset_y, scale):
    """
    屏幕坐标转换为原图坐标
    
    公式: (screen - offset) / scale
    """
    return (int((screen_x - offset_x) / scale),
            int((screen_y - offset_y) / scale))

# 原图坐标 -> 屏幕坐标（渲染）
def original_to_screen(orig_x, orig_y, offset_x, offset_y, scale):
    """
    原图坐标转换为屏幕坐标
    
    公式: orig * scale + offset
    """
    return (int(orig_x * scale) + offset_x,
            int(orig_y * scale) + offset_y)
```

```python
class ProcessSignals(QObject):
    """
    工作线程信号定义
    
    Signals:
        progress(str): 进度消息
        progress_value(int): 进度值 0-100

            - timings: 耗时统计
        error(str): 错误消息

    progress_value = pyqtSignal(int)
    finished = pyqtSignal(list, list, dict)
    error = pyqtSignal(str)
```

### 6.3 处理流程

```python
class ProcessWorker(QThread):
    def __init__(self, image_cv: np.ndarray):
        super().__init__()
        self.image_cv = image_cv
        self.signals = ProcessSignals()
    
    def run(self):
        try:
            timings = {}
            total_start = time.time()
            
            # Phase 1: 云端视觉 (5% -> 55%)
            self.signals.progress_value.emit(5)
            self.signals.progress.emit("云端视觉识别中...")
            s1 = time.time()
            ocr = detect_by_cloud_vision(self.image_cv, self.signals.progress.emit)
            timings["云端视觉"] = time.time() - s1
            self.signals.progress_value.emit(55)
            
            # Phase 2: 规则匹配 (55% -> 75%)
            self.signals.progress.emit("规则匹配中...")
            s2 = time.time()
            rules = detect_by_rules(ocr)
            timings["规则匹配"] = time.time() - s2
            self.signals.progress_value.emit(75)
            
            # Phase 3: AI分析 (75% -> 100%)
            self.signals.progress.emit("AI分析中...")
            s3 = time.time()
            rule_texts = {h["text"] for h in rules}
            remaining = [b for b in ocr if b["text"] not in rule_texts]
            ai = detect_by_ai(remaining, self.signals.progress.emit)
            timings["AI分析"] = time.time() - s3
            timings["总耗时"] = time.time() - total_start
            
            self.signals.progress_value.emit(100)
            self.signals.progress.emit("处理完成")
            self.signals.finished.emit(ocr, rules + ai, timings)
            
        except Exception as e:
            self.signals.error.emit(str(e))
```

```python
    if not CONFIG["ai_enabled"]:
        return []  # AI 禁用，直接返回空
    
    if not CONFIG["llm_api_key"]:
        return []  # 无 API Key，直接返回空
    
    try:
        # 尝试 AI 识别
        return _detect_by_ai_impl(ocr_blocks, progress_cb)
    except Exception as e:
        dev_log(f"AI分析失败，降级为仅规则匹配: {e}", "WARNING")
        return []  # AI 失败，返回空，不影响主流程
```

### 8.1 渲染优化

```python
# 避免重复渲染
class AnnotationCanvas:
    def __init__(self):
        self._render_pending = False
    
    def schedule_render(self):
        """延迟渲染，合并多次更新"""
        if not self._render_pending:
            self._render_pending = True
            QTimer.singleShot(16, self._do_render)  # 约 60fps
    
    def _do_render(self):
        self._render_pending = False
        self._render()
```

### 8.2 图片处理优化

```python
# 图片压缩策略
def preprocess_image(image: np.ndarray) -> np.ndarray:
    """
    预处理图片，限制尺寸
    
    策略:
    1. 最大边不超过 1800px（API 限制）
    2. JPEG 质量 85（平衡大小和质量）
    """
    h, w = image.shape[:2]
    max_side = CONFIG["cloud_image_max_side"]
    
    if max(h, w) > max_side:
        scale = max_side / max(h, w)
        image = cv2.resize(image, (int(w * scale), int(h * scale)))
    
    return image
```
