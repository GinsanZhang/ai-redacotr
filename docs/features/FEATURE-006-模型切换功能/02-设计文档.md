## 3. 详细设计

### 3.1 config.py 改动

新增函数：

```python
def get_config_dir() -> Path:
    """获取配置文件目录"""
    return Path(__file__).parent.parent.parent / "config"

def load_models_config() -> dict:
    """加载模型配置文件"""
    config_path = get_config_dir() / "models.json"
    if config_path.exists():
        with open(config_path, "r", encoding="utf-8") as f:
            return json.load(f)
    return get_default_models_config()

def get_default_models_config() -> dict:
    """获取默认模型配置（硬编码备份）"""
    return {
        "vlm_models": {
            "default": "Qwen/Qwen3-VL-32B-Instruct",
            "models": [{"id": "Qwen/Qwen3-VL-32B-Instruct", "name": "Qwen3-VL 32B", "supports_thinking": True}]
        },
        "llm_models": {
            "default": "Pro/moonshotai/Kimi-K2.5",
            "models": [{"id": "Pro/moonshotai/Kimi-K2.5", "name": "Kimi K2.5"}]
        }
    }

def load_user_settings() -> dict:
    """加载用户设置"""
    settings_path = get_config_dir() / "user_settings.json"
    if settings_path.exists():
        with open(settings_path, "r", encoding="utf-8") as f:
            return json.load(f)
    return {}

def save_user_settings(settings: dict):
    """保存用户设置"""
    config_dir = get_config_dir()
    config_dir.mkdir(exist_ok=True)
    settings_path = config_dir / "user_settings.json"
    with open(settings_path, "w", encoding="utf-8") as f:
        json.dump(settings, f, ensure_ascii=False, indent=2)
```

CONFIG加载逻辑改动：

```python
# 加载模型配置
_models_config = load_models_config()
_user_settings = load_user_settings()

# VLM模型：用户设置 > 环境变量 > 默认值
CONFIG["vlm_model"] = (
    _user_settings.get("vlm_model") or 
    os.getenv("VLM_MODEL") or 
    _models_config["vlm_models"]["default"]
)

# LLM模型：用户设置 > 环境变量 > 默认值
CONFIG["llm_model"] = (
    _user_settings.get("llm_model") or
    os.getenv("LLM_MODEL") or
    _models_config["llm_models"]["default"]
)

# 存储模型列表供UI使用
CONFIG["vlm_models"] = _models_config["vlm_models"]["models"]
CONFIG["llm_models"] = _models_config["llm_models"]["models"]
```

新增回调方法：

```python
def _on_vlm_model_changed(self, idx):
    model_id = self.vlm_model_combo.currentData()
    CONFIG["vlm_model"] = model_id
    self._save_user_settings()

def _on_llm_model_changed(self, idx):
    model_id = self.llm_model_combo.currentData()
    CONFIG["llm_model"] = model_id
    self._save_user_settings()

def _save_user_settings(self):
    from .config import save_user_settings
    settings = {
        "vlm_model": CONFIG["vlm_model"],
        "llm_model": CONFIG["llm_model"],
        "vlm_mode": CONFIG.get("vlm_mode", "fast"),
        "rule_enabled": self.rule_checkbox.isChecked(),
        "llm_enabled": self.llm_checkbox.isChecked()
    }
    save_user_settings(settings)
```

```python
{"model": CONFIG["llm_model"], ...}  # 已是动态值
```

仅需更新 `VLM_MODELS_SUPPORT_THINKING` 为从模型配置中读取：

```python
def get_model_supports_thinking(model_id: str) -> bool:
    """检查模型是否支持深度思考"""
    for model in CONFIG.get("vlm_models", []):
        if model["id"] == model_id:
            return model.get("supports_thinking", False)
    return False
```
