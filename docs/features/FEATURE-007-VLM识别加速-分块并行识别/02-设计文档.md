```python
def prepare_lightweight_image(image: np.ndarray) -> tuple:
    h, w = image.shape[:2]
    scale = min(1.0, 1024.0 / max(h, w))  # 缩放到 1024
    resized = cv2.resize(image, (int(w * scale), int(h * scale))) if scale < 1.0 else image
    # JPEG 质量 75（比正常的 85 更小）
    ok, buf = cv2.imencode(".jpg", resized, [cv2.IMWRITE_JPEG_QUALITY, 75])
    b64 = base64.b64encode(buf.tobytes()).decode("utf-8")
    return b64, resized.shape[1], resized.shape[0]  # 返回新宽高
```

### 2.2 Step 2: 智能分块（按行分组）

```python
def smart_chunking(blocks: list, max_per_chunk: int = 20) -> list:
    """
    按行智能分块
    
    逻辑:
    1. 按 y 坐标排序
    2. 找到所有行的分割点（y 坐标差 > 阈值）
    3. 每行作为一个 chunk
    4. 如果某行 > max_per_chunk，再细分
    """
    if not blocks:
        return []
    
    # 按 y 坐标排序
    sorted_blocks = sorted(blocks, key=lambda b: b["bbox"][1])
    
    # 分行：y 坐标差 > 20 视为不同行
    ROW_THRESHOLD = 20
    rows = []
    current_row = [sorted_blocks[0]]
    
    for block in sorted_blocks[1:]:
        y = block["bbox"][1]
        prev_y = current_row[-1]["bbox"][1]
        if y - prev_y > ROW_THRESHOLD:
            rows.append(current_row)
            current_row = [block]
        else:
            current_row.append(block)
    rows.append(current_row)
    
    # 每行限制 max_per_chunk
    chunks = []
    for row in rows:
        if len(row) <= max_per_chunk:
            chunks.append(row)
        else:
            # 如果某行超过限制，按列继续拆分
            sub_chunks = [row[i:i+max_per_chunk] for i in range(0, len(row), max_per_chunk)]
            chunks.extend(sub_chunks)
    
    return chunks
```

```python
from concurrent.futures import ThreadPoolExecutor, as_completed

def parallel_recognition(image, chunks, progress_cb=None):
    """并行调用 VLM 识别每个 chunk"""
    
    def recognize_chunk(chunk_blocks):
        # 计算 chunk 的边界框
        min_x = min(b["bbox"][0] for b in chunk_blocks)
        min_y = min(b["bbox"][1] for b in chunk_blocks)
        max_x = max(b["bbox"][2] for b in chunk_blocks)
        max_y = max(b["bbox"][3] for b in chunk_blocks)
        
        # 裁剪图像（扩大一点边界）
        padding = 10
        crop_x1 = max(0, min_x - padding)
        crop_y1 = max(0, min_y - padding)
        crop_x2 = min(image.shape[1], max_x + padding)
        crop_y2 = min(image.shape[0], max_y + padding)
        
        cropped = image[crop_y1:crop_y2, crop_x1:crop_x2]
        
        # 调用 VLM 详细识别
        return call_vlm_detailed(cropped, chunk_blocks, crop_x1, crop_y1)
    
    all_results = []
    with ThreadPoolExecutor(max_workers=20) as executor:
        futures = {executor.submit(recognize_chunk, chunk): i for i, chunk in enumerate(chunks)}
        
        for future in as_completed(futures):
            try:
                result = future.result()
                all_results.extend(result)
            except Exception as e:
                dev_log(f"Chunk识别失败: {e}", "ERROR")
    
    return all_results
```

### 2.4 Step 4: 合并结果

```python
def merge_results(all_results: list) -> dict:
    """合并所有结果"""
    
    ocr_blocks = []
    sensitive_hits = []
    
    for result in all_results:
        ocr_blocks.extend(result.get("ocr_blocks", []))
        sensitive_hits.extend(result.get("sensitive_hits", []))
    
    # 按位置排序（先 y 后 x）
    ocr_blocks.sort(key=lambda b: (b["bbox"][1], b["bbox"][0]))
    
    return {
        "ocr_blocks": ocr_blocks,
        "sensitive_hits": sensitive_hits
    }
```

### 5.1 进度追踪

```python
class ChunkProgress:
    def __init__(self, total_chunks):
        self.total = total_chunks
        self.completed = 0
        self.current_chunk_idx = 0
        self.results = []  # 实时存储已完成的块结果

def on_chunk_started(chunk_idx):
    """某块开始识别时调用"""
    # 更新UI：标记该区域为"识别中"状态（黄色边框）
    canvas.set_region_status(chunk_idx, "processing")

def on_chunk_finished(chunk_idx, result):
    """某块识别完成时调用"""
    # 更新进度
    progress.completed += 1
    progress.results.append(result)
    
    # 更新UI：
    # 1. 标记该区域为"已完成"（绿色边框）
    # 2. 渲染已识别文字块到画布
    # 3. 更新进度条 (completed / total)
    # 4. 更新状态文字 "已识别 {completed}/{total} 区域"
    canvas.set_region_status(chunk_idx, "completed")
    canvas.add_blocks(result["ocr_blocks"])
    progress_bar.setValue(int(progress.completed / progress.total * 100))
    status_label.setText(f"已识别 {progress.completed}/{progress.total} 区域")

def on_error(chunk_idx, error):
    """某块识别失败时调用"""
    # 标记该区域为"失败"（红色边框）
    canvas.set_region_status(chunk_idx, "error")
```

## 6. 回退机制

如果分块识别失败（识别不完整），回退到单次完整识别：

```python
def detect_by_vlm_with_fallback(image, progress_cb=None):
    try:
        # 尝试分块并行识别
        result = detect_by_vlm_parallel(image, progress_cb)
        
        # 检查结果是否完整
        if len(result["ocr_blocks"]) < 3:  # 至少应该有3个文字块
            raise Exception("识别结果过少，可能不完整")
        
        return result
    except Exception as e:
        dev_log(f"并行识别失败，回退到单次识别: {e}", "WARNING")
        # 回退到原有的 detect_by_vlm
        return detect_by_vlm(image, progress_cb, include_sensitive=True)
```

## 9. 风险控制

1. **API 限流**: 限制 20 并发，避免触发限制
2. **识别不完整**: 回退机制确保功能可用
3. **坐标偏移**: 裁剪时记录偏移量，正确转换坐标
4. **内存占用**: 并发数控制，避免 OOM
