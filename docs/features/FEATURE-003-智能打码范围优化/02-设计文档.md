#### 颜色与样式
- **标签文字**: 保持原始颜色，清晰可见
- **敏感值区域**: 应用用户选择的打码样式（模糊/像素/纯黑）

```
            ↓

            ↓

视觉呈现:
┌─────────────────────────────────────┐
│ 标签文字│马赛克马赛克马赛克马赛克    │
```

### 1.4 Edge Cases Handling（边界情况处理）

| 场景 | 处理方式 | 视觉结果 |
|------|----------|----------|
| 标签过长 | 按比例计算，确保敏感区至少保留30%宽度 | 标签可能部分可见 |
| 值过短（1-2字符） | 正常打码，最小宽度24px | 小区域马赛克 |
| 无分隔符 | 降级为整段打码 | 与当前行为一致 |
| 多个分隔符 | 使用第一个有效分隔符 | 标签=第一部分 |
| 标签不在白名单 | 降级为整段打码 | 安全优先 |
| 敏感值位置计算偏差 | 添加padding=5%容错 | 略微扩大打码区 |

### 3.1 标签-值分离逻辑

#### 实现位置
`src/redactor/core.py`

#### 核心函数
```python
def split_label_value(text: str) -> tuple:
    """
    分离标签和敏感值
    
    Args:
        text: 完整文字，如 "姓名：张三"
        
    Returns:
        (label, value, separator) 或 (None, text, None)
        
    Examples:
        >>> split_label_value("姓名：张三")
        ("姓名", "张三", "：")
        >>> split_label_value("Name: John")
        ("Name", "John", ": ")
        >>> split_label_value("张三")
        (None, "张三", None)
    """
    # 支持的分隔符（按优先级排序）
    separators = ['：', ': ', ':', ' - ', '-', '  ', ' ']
    
    for sep in separators:
        if sep in text:
            parts = text.split(sep, 1)  # 只分割一次
            if len(parts) == 2:
                label = parts[0].strip()
                value = parts[1].strip()
                
                # 验证标签有效性
                if _is_valid_label(label):
                    return label, value, sep
    
    return None, text, None


def _is_valid_label(label: str) -> bool:
    """
    检查标签是否在白名单中
    
    支持部分匹配，如 "用户姓名" 匹配 "姓名"
    """
    if not label:
        return False
        
    # 转换为小写进行大小写不敏感比较
    label_lower = label.lower()
    
    for valid in VALID_LABELS:
        if valid.lower() in label_lower or label_lower in valid.lower():
            return True
    
    return False
```

### 3.2 敏感区域坐标计算

#### 核心算法
```python
def calculate_sensitive_bbox(
    full_bbox: list, 
    full_text: str, 
    sensitive_text: str,
    separator: str = None
) -> list:
    """
    根据字符比例估算敏感值在文字块中的位置
    
    Args:
        full_bbox: 整个文字块坐标 [x1, y1, x2, y2]
        full_text: 完整文字
        sensitive_text: 敏感值文字
        separator: 分隔符（用于精确定位）
        
    Returns:
        sensitive_bbox: 敏感值区域坐标 [sx1, y1, sx2, y2]
        
    Algorithm:
        1. 基于字符均匀分布假设
        2. 计算敏感值起始字符位置
        3. 按比例映射到bbox坐标
        4. 添加安全padding
    """
    x1, y1, x2, y2 = full_bbox
    total_width = x2 - x1
    total_chars = len(full_text)
    
    # 找到敏感值在文本中的起始索引
    start_idx = full_text.find(sensitive_text)
    if start_idx == -1:
        # 找不到敏感值，降级为整段打码
        return full_bbox
    
    # 计算字符宽度（均匀分布假设）
    char_width = total_width / max(total_chars, 1)
    
    # 计算敏感值坐标
    sensitive_chars = len(sensitive_text)
    sensitive_width = sensitive_chars * char_width
    
    # 起始位置
    sensitive_x1 = x1 + start_idx * char_width
    sensitive_x2 = sensitive_x1 + sensitive_width
    
    # 添加安全padding（5%容错）
    padding = max(2, int(sensitive_width * 0.05))
    sensitive_x1 = max(x1, int(sensitive_x1) - padding)
    sensitive_x2 = min(x2, int(sensitive_x2) + padding)
    
    # 确保最小宽度（至少24px）
    if sensitive_x2 - sensitive_x1 < 24:
        center = (sensitive_x1 + sensitive_x2) // 2
        sensitive_x1 = max(x1, center - 12)
        sensitive_x2 = min(x2, center + 12)
    
    return [sensitive_x1, y1, sensitive_x2, y2]
```

```

sensitive_text = "张三"
separator = "："

字符分布:
┌────┬────┬────┬────┐
│ 姓 │ 名 │ ： │ 张 │ 三 │
└────┴────┴────┴────┘
  0    1    2    3    4
            ↑
       分隔符位置

敏感值起始索引: 3 (字符 "张")
敏感值长度: 2 ("张三")

┌──────────────────────────────┐
│  姓  │  名  │  ：  │  张  │ 三 │
└──────────────────────────────┘
↑                   ↑
x1                 sx1
     <── label ──>
                    <─sensitive─>
                                  ↑
                                 x2
```

```python
def detect_by_rules(ocr_blocks: list) -> list:
    """
    使用正则表达式匹配敏感信息，并计算精确打码区域
    """
    hits = []
    for block in ocr_blocks:
        text = block["text"]
        full_bbox = block["bbox"]
        
        for label, pattern in PATTERNS.items():
            if re.search(pattern, text):
                # 尝试分离标签和值
                label_part, value_part, separator = split_label_value(text)
                
                if label_part and value_part:
                    # 成功分离，计算精确打码区域
                    sensitive_bbox = calculate_sensitive_bbox(
                        full_bbox, text, value_part, separator
                    )
                    hits.append({
                        "bbox": full_bbox,
                        "sensitive_bbox": sensitive_bbox,
                        "label": label,
                        "text": text,
                        "sensitive_text": value_part,
                        "source": "rule"
                    })
                else:
                    # 无法分离，降级为整段打码
                    hits.append({
                        "bbox": full_bbox,
                        "label": label,
                        "text": text,
                        "source": "rule"
                    })
                break
    return hits
```

### 3.4 打码应用集成

#### 修改 apply_mosaic()
```python
def apply_mosaic(
    image: np.ndarray, 
    bbox: list, 
    style: str = "blur", 
    strength: int = 20,
    sensitive_bbox: list = None
) -> np.ndarray:
    """
    对指定区域应用马赛克
    
    Args:
        image: 原始图像
        bbox: 边界框（兼容旧接口）
        style: 打码样式
        strength: 打码强度
        sensitive_bbox: 敏感值区域（可选，优先使用）
    """
    # 优先使用 sensitive_bbox，降级使用 bbox
    target_bbox = sensitive_bbox if sensitive_bbox else bbox
    
    x1, y1, x2, y2 = target_bbox
    
    # 添加padding并裁剪到图像边界
    p = 4
    x1 = max(0, x1 - p)
    y1 = max(0, y1 - p)
    x2 = min(image.shape[1], x2 + p)
    y2 = min(image.shape[0], y2 + p)
    
    if x2 <= x1 or y2 <= y1:
        return image
    
    region = image[y1:y2, x1:x2]
    
    # 应用打码样式
    if style == "blur":
        k = max(strength | 1, 21)
        image[y1:y2, x1:x2] = cv2.GaussianBlur(region, (k, k), 0)
    elif style == "pixel":
        h, w = region.shape[:2]
        block = max(strength // 3, 8)
        small = cv2.resize(region, (max(1, w // block), max(1, h // block)), 
                          interpolation=cv2.INTER_LINEAR)
        image[y1:y2, x1:x2] = cv2.resize(small, (w, h), 
                                        interpolation=cv2.INTER_NEAREST)
    elif style == "block":
        image[y1:y2, x1:x2] = 0
    
    return image
```

### 3.5 Fallback Strategy（降级策略）

#### 降级条件
1. 无法识别分隔符
2. 标签不在白名单中
3. 敏感值文本找不到
4. 计算出的敏感区域过小（< 10% 总宽度）
5. 计算出的敏感区域越界

#### 降级行为
```python
def calculate_sensitive_bbox(full_bbox, full_text, sensitive_text, separator=None):
    """计算敏感区域，带完整降级逻辑"""
    
    # 基础验证
    if not sensitive_text or not full_text:
        return full_bbox
    
    # 查找敏感值位置
    start_idx = full_text.find(sensitive_text)
    if start_idx == -1:
        return full_bbox  # 降级1: 找不到敏感值
    
    # 坐标计算
    x1, y1, x2, y2 = full_bbox
    total_width = x2 - x1
    total_chars = len(full_text)
    
    if total_chars == 0:
        return full_bbox  # 降级2: 空文本
    
    char_width = total_width / total_chars
    sensitive_chars = len(sensitive_text)
    
    sensitive_x1 = x1 + start_idx * char_width
    sensitive_x2 = sensitive_x1 + sensitive_chars * char_width
    
    # 验证合理性
    sensitive_width = sensitive_x2 - sensitive_x1
    if sensitive_width < total_width * 0.1:
        return full_bbox  # 降级3: 敏感区过小（<10%）
    
    if sensitive_x1 < x1 or sensitive_x2 > x2:
        return full_bbox  # 降级4: 坐标越界
    
    return [int(sensitive_x1), y1, int(sensitive_x2), y2]
```

## 5. 风险与应对措施

### 5.1 技术风险

| 风险 | 可能性 | 影响 | 应对措施 |
|------|--------|------|----------|
| 字符比例估算不准确 | 中 | 打码位置偏移 | 1. 添加5%容错padding<br>2. 提供手动调整功能 |
| 复杂格式无法分界 | 高 | 降级为整段打码 | 接受降级，不破坏原有功能 |
| 中文字符宽度不均 | 中 | 坐标计算偏差 | 实测调整，增大容错空间 |

### 5.2 兼容性风险

| 风险 | 可能性 | 影响 | 应对措施 |
|------|--------|------|----------|
| 旧Region结构不兼容 | 低 | 程序报错 | sensitive_bbox为可选字段 |
| 手动区域丢失功能 | 低 | 无法打码 | 手动区域继续使用bbox |
