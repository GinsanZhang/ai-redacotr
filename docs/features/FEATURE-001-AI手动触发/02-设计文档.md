## 2. 架构设计

### 2.1 流程变更对比

**当前流程（自动）**:
```
截图/打开图片 -> 自动启动ProcessWorker -> OCR -> 规则 -> AI -> 显示结果
```

**新流程（手动）**:
```
截图/打开图片 -> 仅加载图片到画布 -> 等待用户点击 -> 启动ProcessWorker -> OCR -> 规则 -> AI -> 显示结果
```

### 2.3 关键变更点

| 文件 | 方法 | 变更 |
|------|------|------|
| main_window.py | `_process_image()` | 不再自动创建ProcessWorker，仅加载图片并启用按钮 |
| main_window.py | `_make_toolbar()` | 添加btnStartRecognition按钮 |

#### 3.2.2 main_window.py - _process_image()

修改后的逻辑：

```python
def _process_image(self, image_cv):
    """处理图片 - 仅加载到画布，不启动识别"""
    self.current_cv = image_cv
    self.worker = None  # 不创建worker
    
    # 重置保存/复制按钮
    self.btn_save.setEnabled(False)
    self.btn_copy.setEnabled(False)
    
    # 加载图片到画布（不带任何识别结果）
    self.canvas.load_image(image_cv, [])
    
    # 启用开始识别按钮
    self.btn_start_recognition.setEnabled(True)
    self.btn_start_recognition.setText("🔍 开始识别")
    
    # 重置进度和状态
    self.progress_bar.setValue(0)
    self.progress_label.setText("等待识别")
    self.stats_label.setText("图片已加载，点击"开始识别"进行隐私检测")
    self.status.showMessage("图片已加载，点击"开始识别"进行隐私检测")
```

#### 3.2.3 main_window.py - _start_recognition() (新增)

```python
def _start_recognition(self):
    """用户点击开始识别按钮，启动识别流程"""
    if self.current_cv is None:
        return
    
    # 禁用按钮，防止重复点击
    self.btn_start_recognition.setEnabled(False)
    self.btn_start_recognition.setText("识别中...")
    
    # 禁用保存/复制按钮
    self.btn_save.setEnabled(False)
    self.btn_copy.setEnabled(False)
    
    # 重置进度和状态
    self.progress_bar.setValue(0)
    self.stats_label.setText("识别中...")
    self.status.showMessage("开始识别...")
    
    # 创建并启动worker
    from .ui import ProcessWorker
    self.worker = ProcessWorker(self.current_cv)
    self.worker.signals.progress.connect(self.progress_label.setText)
    self.worker.signals.progress_value.connect(self.progress_bar.setValue)
    self.worker.signals.finished.connect(self._on_process_finished)
    self.worker.signals.error.connect(self._on_process_error)
    self.worker.start()
```

#### 3.2.4 main_window.py - _on_process_finished()

```python
def _on_process_finished(self, ocr, hits, timings):
    """识别完成回调"""
    self.canvas.load_image(self.current_cv, hits)
    
    # 启用保存/复制按钮
    self.btn_save.setEnabled(True)
    self.btn_copy.setEnabled(True)
    
    # 进度和状态
    self.progress_bar.setValue(100)
    self.progress_label.setText("处理完成")
    
    # 统计信息
    ai_hits = len([h for h in hits if h.get("source") == "ai"])
    rule_hits = len([h for h in hits if h.get("source") == "rule"])
    self.stats_label.setText(f"文字块: {len(ocr)}\n规则命中: {rule_hits}\nAI命中: {ai_hits}\n合计打码: {len(hits)}")
    self.timing_label.setText(f"耗时: {timings.get('总耗时', 0):.2f}s")
    self.status.showMessage(f"处理完成，共识别 {len(hits)} 处敏感信息")
    
    # 重新启用开始识别按钮，文字改为"重新识别"
    self.btn_start_recognition.setEnabled(True)
    self.btn_start_recognition.setText("🔍 重新识别")
```

#### 3.2.5 main_window.py - _on_process_error()

```python
def _on_process_error(self, msg):
    """识别错误回调"""
    self.btn_save.setEnabled(False)
    self.btn_copy.setEnabled(False)
    self.progress_bar.setValue(0)
    self.progress_label.setText("处理失败")
    self.stats_label.setText("识别失败，请检查 Key、网络或服务权限")
    self.timing_label.setText("阶段耗时：-")
    self.status.showMessage(f"错误: {msg}")
    
    # 错误时也重新启用按钮，允许重试
    self.btn_start_recognition.setEnabled(True)
    self.btn_start_recognition.setText("🔍 重新识别")
```
